# 前后台用户分离改造计划

> 目标：对标 BuildAdmin，实现前台会员（Member）与后台管理员（Admin）完全分离的认证体系

---

## 一、改造概览

### 1.1 命名规范（重要）

| 概念 | 后台管理员 | 前台会员 |
|------|-----------|---------|
| **英文名** | Admin | Member |
| **数据库表** | `admin_user` | `member` |
| **后端路由前缀** | `/admin/**` | `/member/**` |
| **Token Header** | `Authorization: Bearer xxx` | `Xy-User-Token: xxx` |
| **前端 Store** | `userStore`（保持原有） | `memberStore`（新增） |
| **前端 API** | `adminRequest` | `memberRequest` |
| **前端视图目录** | `views/backend/` | `views/frontend/` |
| **前端路由路径** | `/login`, `/dashboard/*` | `/`, `/user/*` |

### 1.2 当前架构 vs 目标架构

| 对比项 | 当前 xygo | 目标架构 |
|--------|-----------|----------|
| 用户表 | 单表 `admin_user` | 双表：`admin_user` + `member` |
| Token 存储 | 单一 `userStore` | 双 Store：`userStore` + `memberStore` |
| Token Header | `Authorization: Bearer xxx` | Admin: `Authorization`<br>Member: `Xy-User-Token` |
| 后端路由前缀 | `/admin/**` | Admin: `/admin/**`<br>Member: `/member/**` |
| 前端登录入口 | `/login` (后台) | Admin: `/login`<br>Member: `/user/login` |

### 1.3 改造范围

```
┌─────────────────────────────────────────────────────────────┐
│                        后端改造                              │
├─────────────────────────────────────────────────────────────┤
```

### 3.3 会员认证中间件 `internal/middleware/member_auth.go`

```go
package middleware

import (
    "github.com/gogf/gf/v2/errors/gerror"
    "github.com/gogf/gf/v2/net/ghttp"
    
    "xygo/internal/consts"
    "xygo/internal/library/contexts"
    "xygo/internal/library/token"
)

// MemberAuth 会员接口鉴权中间件
// - 从 Header 获取 Xy-User-Token
// - 验证通过后注入会员信息到上下文
func MemberAuth(r *ghttp.Request) {
    path := r.URL.Path
    
    // 初始化自定义上下文
    customCtx := &contexts.Context{
        Module: "member",
    }
    contexts.Init(r, customCtx)
    
    // 白名单接口放行（登录、注册等）
    whitelist := []string{
        "/member/auth/login",
        "/member/auth/register",
        "/member/auth/captcha",
    }
    for _, p := range whitelist {
        if path == p {
            r.Middleware.Next()
            return
        }
    }
    
    // 获取会员 Token（使用 Xy-User-Token 头）
    tokenStr := r.Header.Get("Xy-User-Token")
    if tokenStr == "" {
        r.SetError(gerror.NewCode(consts.CodeNotAuthorized, "请先登录"))
        return
    }
    
    // 解析 Token（使用 member 应用名）
    memberUser, err := token.Parse(r.Context(), "member", tokenStr)
    if err != nil {
        r.SetError(gerror.NewCode(consts.CodeNotAuthorized, "登录已过期，请重新登录"))
        return
    }
    
    // 注入会员信息到上下文
    contexts.SetUser(r.Context(), memberUser)
    
    r.Middleware.Next()
}
```

## 五、前端改造 - 阶段 4：Store 与 HTTP

### 5.1 新增 memberStore `store/modules/member.ts`

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { CacheKey } from '@/utils/cache/keys'
import { StorageType } from '@/utils/cache/types'
import { useCache } from '@/hooks/core/useCache'

// 会员信息类型
interface MemberInfo {
  id: number
  username: string
  nickname: string
  avatar: string
  mobile: string
  email: string
  gender: number
  level: number
  score: number
}

// 缓存 Key
const MEMBER_INFO_KEY = 'memberInfo'
const MEMBER_TOKEN_KEY = 'memberToken'

export const useMemberStore = defineStore('memberStore', () => {
  const cache = useCache(StorageType.LOCAL)
  
  // 状态
  const token = ref<string>(cache.get(MEMBER_TOKEN_KEY) || '')
  const memberInfo = ref<MemberInfo | null>(cache.get(MEMBER_INFO_KEY) || null)
  const isLogin = ref<boolean>(!!token.value)
  
  // 设置 Token
  const setToken = (newToken: string) => {
    token.value = newToken
    isLogin.value = !!newToken
    cache.set(MEMBER_TOKEN_KEY, newToken)
  }
  
  // 设置会员信息
  const setMemberInfo = (info: MemberInfo) => {
    memberInfo.value = info
    cache.set(MEMBER_INFO_KEY, info)
  }
  
  // 获取 Token
  const getToken = (): string => {
    return token.value
  }
  
  // 获取会员信息
  const getMemberInfo = (): MemberInfo | null => {
    return memberInfo.value
  }
  
  // 退出登录
  const logout = () => {
    token.value = ''
    memberInfo.value = null
    isLogin.value = false
    cache.remove(MEMBER_TOKEN_KEY)
    cache.remove(MEMBER_INFO_KEY)
  }
  
  return {
    token,
    memberInfo,
    isLogin,
    setToken,
    setMemberInfo,
    getToken,
    getMemberInfo,
    logout
  }
})
```

### 5.2 HTTP 拦截器改造 `utils/http/index.ts`

```typescript
// 请求拦截器改造
service.interceptors.request.use((config: AxiosRequestConfig) => {
  const userStore = useUserStore()       // 后台管理员
  const memberStore = useMemberStore()   // 前台会员
  
  // 根据请求 URL 决定携带哪个 Token
  const url = config.url || ''
  
  if (url.startsWith('/member')) {
    // 会员接口：使用 Xy-User-Token
    const memberToken = memberStore.getToken()
    if (memberToken) {
      config.headers['Xy-User-Token'] = memberToken
    }
  } else if (url.startsWith('/admin')) {
    // 管理员接口：使用 Authorization Bearer
    const adminToken = userStore.getToken()
    if (adminToken) {
      config.headers['Authorization'] = `Bearer ${adminToken}`
    }
  }
  
  return config
})

// 响应拦截器改造（处理双端 401）
service.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      const url = error.config?.url || ''
      
      if (url.startsWith('/member')) {
        // 会员 Token 失效
        const memberStore = useMemberStore()
        memberStore.logout()
        router.push('/user/login')
      } else if (url.startsWith('/admin')) {
        // 管理员 Token 失效
        const userStore = useUserStore()
        userStore.logOut()
        router.push('/login')
      }
    }
    return Promise.reject(error)
  }
)
```

### 6.1 路由守卫改造

```typescript
// 区分前后台权限校验
function handleLoginStatus(to, next) {
  const userStore = useUserStore()       // 后台
  const memberStore = useMemberStore()   // 前台
  
  // 前台路由
  if (isFrontendRoute(to.path)) {
    if (requiresAuth(to) && !memberStore.isLogin) {
      next({ name: 'UserLogin', query: { redirect: to.fullPath } })
      return false
    }
    return true
  }
  
  // 后台路由
  if (!userStore.isLogin && to.path !== '/login') {
    next({ name: 'Login', query: { redirect: to.fullPath } })
    return false
  }
  
  return true
}
```

## 八、Token Header 规范

| 端 | Header 名 | 格式 | 示例 |
|----|-----------|------|------|
| 后台管理员 | `Authorization` | `Bearer {token}` | `Bearer eyJhbGci...` |
| 前台会员 | `Xy-User-Token` | `{token}` | `eyJhbGci...` |

## 十、备注

1. **数据隔离**：管理员表 `admin_user` 和会员表 `member` 完全独立，ID 可以重复
2. **Token 隔离**：两套 Token 使用不同的缓存 Key 前缀（`xygo:token:admin:*` vs `xygo:token:member:*`）
3. **权限隔离**：后台菜单 `admin_menu` 和前台菜单 `member_menu` 独立管理
4. **前后台可同时登录**：不会互相踢出
