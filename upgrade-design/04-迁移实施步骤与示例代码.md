## 阶段 2：Request 封装

### 2.1 创建分层请求实例

```typescript
// src/utils/request/index.ts

import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig } from 'axios'

// 创建请求实例工厂
function createRequest(baseURL: string): AxiosInstance {
  const instance = axios.create({
    baseURL,
    timeout: 30000,
    headers: {
      'Content-Type': 'application/json'
    }
  })
  
  // 请求拦截器
  instance.interceptors.request.use(
    config => {
      const token = localStorage.getItem('token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      return config
    },
    error => Promise.reject(error)
  )
  
  // 响应拦截器
  instance.interceptors.response.use(
    response => response.data,
    error => {
      // 统一错误处理
      return Promise.reject(error)
    }
  )
  
  return instance
}

// 后台请求实例
export const adminRequest = createRequest('/admin')

// 前台请求实例
export const siteRequest = createRequest('/site')

// 默认导出后台请求
export default adminRequest
```

### 2.2 API 文件示例

```typescript
// src/api/backend/system/user.ts

import { adminRequest } from '@/utils/request'

export interface UserInfo {
  id: number
  username: string
  nickname: string
  email: string
  phone: string
  status: number
}

export interface UserListParams {
  page: number
  pageSize: number
  username?: string
  status?: number
}

// 获取用户列表
export function getUserList(params: UserListParams) {
  return adminRequest.get<UserInfo[]>('/user/list', { params })
}

// 获取用户详情
export function getUserDetail(id: number) {
  return adminRequest.get<UserInfo>(`/user/detail/${id}`)
}

// 创建用户
export function createUser(data: Partial<UserInfo>) {
  return adminRequest.post('/user/create', data)
}

// 更新用户
export function updateUser(id: number, data: Partial<UserInfo>) {
  return adminRequest.put(`/user/update/${id}`, data)
}

// 删除用户
export function deleteUser(id: number) {
  return adminRequest.delete(`/user/delete/${id}`)
}
```

## 阶段 4：Store 拆分

### 4.1 后台用户 Store

```typescript
// src/stores/backend/user.ts

import { defineStore } from 'pinia'
import { getUserInfo, login, logout } from '@/api/backend/auth/login'

interface UserState {
  token: string
  userInfo: UserInfo | null
  permissions: string[]
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: localStorage.getItem('token') || '',
    userInfo: null,
    permissions: []
  }),
  
  actions: {
    async login(username: string, password: string) {
      const res = await login({ username, password })
      this.token = res.data.token
      localStorage.setItem('token', res.data.token)
    },
    
    async fetchUserInfo() {
      const res = await getUserInfo()
      this.userInfo = res.data.user
      this.permissions = res.data.permissions
    },
    
    async logout() {
      await logout()
      this.token = ''
      this.userInfo = null
      this.permissions = []
      localStorage.removeItem('token')
    }
  }
})
```

## 执行顺序

1. **创建目录结构**（不影响现有代码）
2. **修改 Request 封装**（添加 adminRequest/siteRequest）
3. **添加路径别名**
4. **逐步迁移 API 文件**
5. **逐步迁移 Views 文件**
6. **重构路由模块**
7. **重构 Store 模块**
8. **创建插件管理器**
9. **测试验证**

## 注意事项

1. **分批迁移**：每次迁移一个功能模块，保证可回退
2. **路径修复**：迁移后需批量更新 import 路径
3. **测试覆盖**：每个模块迁移后需完整测试
4. **Git 提交**：每个阶段完成后提交，便于回溯
