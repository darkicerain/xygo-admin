# 快速开始：新增模块

本文档演示如何在 XYGo 中新增一个完整的 CRUD 模块（以"通知公告"为例）。

---

## 一、创建数据表

```sql
-- 1. 创建表
CREATE TABLE `xy_admin_notice` (
    `id` bigint unsigned NOT NULL AUTO_INCREMENT COMMENT 'ID',
    `title` varchar(100) NOT NULL COMMENT '标题',
    `content` text COMMENT '内容',
    `type` tinyint NOT NULL DEFAULT '1' COMMENT '类型：1通知 2公告',
    `status` tinyint NOT NULL DEFAULT '1' COMMENT '状态：0禁用 1启用',
    `create_by` bigint unsigned DEFAULT '0' COMMENT '创建人',
    `create_time` datetime DEFAULT NULL COMMENT '创建时间',
    `update_time` datetime DEFAULT NULL COMMENT '更新时间',
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='通知公告';
```

## 三、定义输入输出模型

创建 `internal/model/input/adminin/notice.go`：

```go
package adminin

import "github.com/gogf/gf/v2/os/gtime"

// NoticeListInp 通知列表查询参数
type NoticeListInp struct {
	Page     int    `json:"page" v:"required|min:1"`
	PageSize int    `json:"pageSize" v:"required|min:1|max:100"`
	Title    string `json:"title"`    // 标题（模糊搜索）
	Type     int    `json:"type"`     // 类型：1通知 2公告
	Status   int    `json:"status"`   // 状态：0禁用 1启用
}

// NoticeListItem 通知列表项
type NoticeListItem struct {
	Id         uint64      `json:"id"`
	Title      string      `json:"title"`
	Content    string      `json:"content"`
	Type       int         `json:"type"`
	Status     int         `json:"status"`
	CreateBy   uint64      `json:"createBy"`
	CreateTime *gtime.Time `json:"createTime"`
	UpdateTime *gtime.Time `json:"updateTime"`
}

// NoticeListModel 通知列表响应（分页）
type NoticeListModel struct {
	List  []NoticeListItem `json:"list"`
	Total int              `json:"total"`
}

// NoticeDetailInp 通知详情查询参数
type NoticeDetailInp struct {
	Id uint64 `json:"id" v:"required"`
}

// NoticeSaveInp 保存通知（新增/编辑）
type NoticeSaveInp struct {
	Id      uint64 `json:"id"`      // ID（为0时新增，否则编辑）
	Title   string `json:"title" v:"required|length:1,100"`
	Content string `json:"content"`
	Type    int    `json:"type" v:"required|in:1,2"`
	Status  int    `json:"status" v:"required|in:0,1"`
}

// NoticeDeleteInp 删除通知
type NoticeDeleteInp struct {
	Id uint64 `json:"id" v:"required"`
}
```

## 四、编写 Logic 层

创建 `internal/logic/notice/notice.go`：

```go
package notice

import (
	"context"
	"xygo/internal/dao"
	"xygo/internal/model/entity"
	"xygo/internal/model/input/adminin"
	"xygo/internal/service"

	"github.com/gogf/gf/v2/errors/gerror"
)

type sAdminNotice struct{}

func init() {
	service.RegisterAdminNotice(New())
}

func New() *sAdminNotice {
	return &sAdminNotice{}
}

// List 获取通知列表
func (s *sAdminNotice) List(ctx context.Context, in *adminin.NoticeListInp) (list []adminin.NoticeListItem, total int, err error) {
	model := dao.AdminNotice.Ctx(ctx)

	// 过滤：标题（模糊搜索）
	if in.Title != "" {
		model = model.WhereLike("title", "%"+in.Title+"%")
	}

	// 过滤：类型
	if in.Type > 0 {
		model = model.Where("type", in.Type)
	}

	// 过滤：状态
	if in.Status == 0 || in.Status == 1 {
		model = model.Where("status", in.Status)
	}

	// 统计总数
	count, err := model.Clone().Count()
	if err != nil {
		return nil, 0, err
	}

	// 分页查询
	var records []adminin.NoticeListItem
	err = model.
		Page(in.Page, in.PageSize).
		OrderDesc("create_time").
		Scan(&records)

	return records, count, err
}

// Detail 获取通知详情
func (s *sAdminNotice) Detail(ctx context.Context, id uint64) (*adminin.NoticeListItem, error) {
	var item adminin.NoticeListItem
	err := dao.AdminNotice.Ctx(ctx).Where("id", id).Scan(&item)
	if err != nil {
		return nil, err
	}
	if item.Id == 0 {
		return nil, gerror.New("通知不存在")
	}
	return &item, nil
}

// Save 保存通知（新增/编辑）
func (s *sAdminNotice) Save(ctx context.Context, in *adminin.NoticeSaveInp) (uint, error) {
	if in.Id > 0 {
		// 编辑
		_, err := dao.AdminNotice.Ctx(ctx).
			Where("id", in.Id).
			Data(entity.AdminNotice{
				Title:   in.Title,
				Content: in.Content,
				Type:    in.Type,
				Status:  in.Status,
			}).
			Update()
		return uint(in.Id), err
	}

	// 新增
	id, err := dao.AdminNotice.Ctx(ctx).
		Data(entity.AdminNotice{
			Title:   in.Title,
			Content: in.Content,
			Type:    in.Type,
			Status:  in.Status,
		}).
		InsertAndGetId()
	return uint(id), err
}

// Delete 删除通知
func (s *sAdminNotice) Delete(ctx context.Context, id uint64) error {
	_, err := dao.AdminNotice.Ctx(ctx).Where("id", id).Delete()
	return err
}
```

```go
import (

// NoticeListReq 通知列表请求
type NoticeListReq struct {
	g.Meta   `path:"/notice/list" method:"post" tags:"通知管理" summary:"获取通知列表"`

	PageSize int    `json:"pageSize" v:"required|min:1|max:100"`
	Title    string `json:"title"`
	Type     int    `json:"type"`
	Status   int    `json:"status"`

type NoticeListRes struct {
	List  []adminin.NoticeListItem `json:"list"`
	Total int                      `json:"total"`
}

// NoticeDetailReq 通知详情请求

	g.Meta `path:"/notice/detail" method:"post" tags:"通知管理" summary:"获取通知详情"`
	Id     uint64 `json:"id" v:"required"`
}

type NoticeDetailRes struct {
	*adminin.NoticeListItem
}

// NoticeSaveReq 保存通知请求
type NoticeSaveReq struct {
	g.Meta  `path:"/notice/save" method:"post" tags:"通知管理" summary:"保存通知"`

	Content string `json:"content"`
	Type    int    `json:"type" v:"required|in:1,2"`
```

## 七、创建 Controller

创建 `internal/controller/admin/notice.go`：

```go
package admin

import (
	"context"
	"xygo/api/admin"
	"xygo/internal/model/input/adminin"
	"xygo/internal/service"
)

var AdminNotice = cAdminNotice{}

type cAdminNotice struct{}

// List 获取通知列表
func (c *cAdminNotice) List(ctx context.Context, req *admin.NoticeListReq) (res *admin.NoticeListRes, err error) {
	list, total, err := service.AdminNotice().List(ctx, &adminin.NoticeListInp{
		Page:     req.Page,
		PageSize: req.PageSize,
		Title:    req.Title,
		Type:     req.Type,
		Status:   req.Status,
	})

	if err != nil {
		return nil, err
	}

	return &admin.NoticeListRes{
		List:  list,
		Total: total,
	}, nil
}

// Detail 获取通知详情
func (c *cAdminNotice) Detail(ctx context.Context, req *admin.NoticeDetailReq) (res *admin.NoticeDetailRes, err error) {
	detail, err := service.AdminNotice().Detail(ctx, req.Id)
	if err != nil {
		return nil, err
	}

	return &admin.NoticeDetailRes{NoticeListItem: detail}, nil
}

// Save 保存通知
func (c *cAdminNotice) Save(ctx context.Context, req *admin.NoticeSaveReq) (res *admin.NoticeSaveRes, err error) {
	id, err := service.AdminNotice().Save(ctx, &adminin.NoticeSaveInp{
		Id:      req.Id,
		Title:   req.Title,
		Content: req.Content,
		Type:    req.Type,
		Status:  req.Status,
	})

	if err != nil {
		return nil, err
	}

	return &admin.NoticeSaveRes{Id: id}, nil
}

// Delete 删除通知
func (c *cAdminNotice) Delete(ctx context.Context, req *admin.NoticeDeleteReq) (res *admin.NoticeDeleteRes, err error) {
	err = service.AdminNotice().Delete(ctx, req.Id)
	return &admin.NoticeDeleteRes{}, err
}
```

## 总结

**新增模块的完整步骤**：

1. ✅ 创建数据表（SQL）
2. ✅ 生成 DAO 层（`.\gen.ps1 dao`）
3. ✅ 定义输入输出模型（`adminin/notice.go`）
4. ✅ 编写 Logic 层（`logic/notice/notice.go`）
5. ✅ 生成 Service 接口（`.\gen.ps1 service`）
6. ✅ 定义 API 结构（`api/admin/admin_notice.go`）
7. ✅ 创建 Controller（`controller/admin/notice.go`）
8. ✅ 注册路由（`cmd/cmd.go`）
9. ✅ 定义字段权限（`field/admin_notice.go`，可选）
10. ✅ 编译运行测试

**核心原则**：
- DAO 层**自动生成**，不要手写
- Logic 层**手写**，实现业务逻辑
- Service 层**自动生成**，不要手写
- Controller 层**手写**，处理 HTTP 请求
- `logic.go` **自动生成**，无需手动维护
